<div id="content"></div>

<script src="https://cdn.jsdelivr.net/npm/marked/marked.min.js"></script>

<script>
  (function () {

    // ---------------- CONFIG ----------------

    const API_BASE =
      "https://api.github.com/repos/janwilmans/publish/contents/";

    const DEFAULT_DOC =
      "Accu on sea 2026/Accu on Sea 2026 Notes.md";

    const TOKEN = "YOUR_GITHUB_TOKEN_HERE"; // optional but recommended

    // ---------------- STATE ----------------

    let lastFileSha = null;
    let lastScrollRatio = 0;

    // ---------------- STYLE ----------------

    const style = document.createElement("style");
    style.innerHTML = `
    #content {
      max-width: 900px;
      margin: 40px auto;
      padding: 24px;

      background: #fff;
      border: 1px solid #ddd;
      border-radius: 12px;

      box-shadow: 0 6px 20px rgba(0,0,0,0.08);

      font-size: 16px;
      line-height: 1.6;
      text-align: left;
    }

    #md-refresh-btn {
      display: inline-block;
      margin-bottom: 16px;
      padding: 8px 12px;

      border: 1px solid #1d4ed8;
      border-radius: 8px;

      background: #2563eb;
      color: white;

      cursor: pointer;
      font-size: 14px;
      font-weight: 500;

      transition: background 0.2s ease, transform 0.1s ease;
    }

    #md-refresh-btn:hover {
      background: #1d4ed8;
    }

    #md-refresh-btn:active {
      transform: scale(0.98);
    }

    #content ul,
    #content ol {
      padding-left: 1.5em !important;
      margin-left: 1em !important;
    }

    #content pre {
      overflow-x: auto;
    }
  `;
    document.head.appendChild(style);

    // ---------------- HELPERS ----------------

    function getDoc() {
      const params = new URLSearchParams(location.search);
      return params.get("doc") || DEFAULT_DOC;
    }

    // normalize encoded paths safely
    function normalizeHref(href) {
      try {
        return decodeURIComponent(href);
      } catch (e) {
        return href;
      }
    }

    // filesystem-style resolver
    function resolvePath(basePath, relativePath) {

      const baseParts = basePath.split("/");
      baseParts.pop(); // remove filename

      const relParts = relativePath.split("/");

      for (const part of relParts) {

        if (part === "..") {
          baseParts.pop();
        } else if (part === "." || part === "") {
          continue;
        } else {
          baseParts.push(part);
        }
      }

      return baseParts.join("/");
    }

    // safe GitHub API encoding (NO double encoding)
    function safeEncodePath(path) {
      return path
        .split("/")
        .map(segment => decodeURIComponent(segment))
        .map(segment => encodeURIComponent(segment))
        .join("/");
    }

    function apiUrl(doc) {
      return API_BASE + safeEncodePath(doc);
    }

    function ghHeaders() {
      const h = {
        Accept: "application/vnd.github+json"
      };

      if (TOKEN && TOKEN !== "YOUR_GITHUB_TOKEN_HERE") {
        h.Authorization = `Bearer ${TOKEN}`;
      }

      return h;
    }

    function saveScroll() {
      const max =
        document.documentElement.scrollHeight - window.innerHeight;

      lastScrollRatio = max > 0 ? window.scrollY / max : 0;
    }

    function restoreScroll() {
      const max =
        document.documentElement.scrollHeight - window.innerHeight;

      window.scrollTo(0, max * lastScrollRatio);
    }

    function rewriteLinks(docPath) {
      document.querySelectorAll("#content a").forEach(a => {
        let href = a.getAttribute("href");

        if (!href || href.startsWith("http") || href.startsWith("#")) return;

        href = normalizeHref(href);

        // BACKREFERENCE SUPPORT HERE
        const targetPath = resolvePath(docPath, href);

        a.href =
          location.pathname +
          "?doc=" +
          encodeURIComponent(targetPath);
      });
    }

    // ---------------- BASE64 DECODER ----------------

    function decodeBase64Unicode(base64) {
      const binary = atob(base64);
      const bytes = Uint8Array.from(binary, c => c.charCodeAt(0));
      return new TextDecoder("utf-8").decode(bytes);
    }

    // ---------------- LOAD ----------------

    async function loadMarkdown(force = false) {

      try {

        const doc = getDoc();
        const url = apiUrl(doc);

        console.log("Fetching API:", url);

        const meta = await fetch(url, {
          headers: ghHeaders()
        }).then(r => r.json());

        if (!meta || meta.message) {
          throw new Error(meta?.message || "GitHub API error");
        }

        if (!meta.sha) {
          throw new Error("Missing SHA in response");
        }

        // &#x1f680; change detection via SHA
        if (!force && meta.sha === lastFileSha) {
          console.log("No changes detected");
          return;
        }

        lastFileSha = meta.sha;

        if (!meta.content) {
          throw new Error("Missing Base64 content");
        }

        const md = decodeBase64Unicode(meta.content);

        saveScroll();

        document.getElementById("content").innerHTML =
          renderCard(md);

        rewriteLinks(doc);

        requestAnimationFrame(restoreScroll);

        console.log("Rendered:", doc);

      } catch (err) {

        console.error(err);

        document.getElementById("content").innerHTML =
          "<pre style='color:red'>" + err.message + "</pre>";
      }
    }

    // ---------------- UI ----------------

    function renderCard(md) {
      return `
      <button id="md-refresh-btn" class="hidden">&#x1f504; Refresh</button>
      <div id="md-body">
        ${marked.parse(md)}
      </div>
    `;
    }

    // ---------------- EVENTS ----------------

    document.addEventListener("click", (e) => {
      if (e.target && e.target.id === "md-refresh-btn") {
        loadMarkdown(true);
      }
    });

    // ---------------- START ----------------

    loadMarkdown();

  })();
</script>
